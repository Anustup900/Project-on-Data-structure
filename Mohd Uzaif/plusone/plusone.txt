class Solution {
public:
    vector<int> plusOne(vector<int>& nums) {
        long long int n=0;
        for(long long int i=0;i<nums.size();i++)
        {
            n=n*10+nums[i];
        }
        nums.clear();
        n=n+1;
        while(n!=0)
        {
            nums.insert(nums.begin(),n%10);
            n=n/10;
        }
        return nums;
    }
};
